01. Whisper Voice

Large Language Models for Voice Control​

This chapter explores the integration of Large Language Models (LLMs) with voice control systems, focusing on technologies like OpenAI's Whisper for robust speech-to-text transcription and subsequent processing by advanced LLMs. It covers the principles of voice command recognition, natural language understanding (NLU), and the generation of appropriate robotic responses or actions.

Overview of Voice-Language-Action (VLA) Models​

VLA models combine speech recognition, language understanding, and action grounding to enable natural voice control of robots.

Pipeline:

Voice Input → Speech-to-Text (Whisper) → Language Understanding (LLM) → Action Grounding → Robot Execution

Components​

1. Speech Recognition (Whisper)

  • Converts audio to text
  • Robust to accents, noise, multilingual

2. Language Understanding (LLM)

  • Interprets user intent
  • Handles ambiguity and context
  • Generates actionable commands

3. Action Grounding

  • Maps language to robot primitives
  • Validates feasibility
  • Generates motion plans

4. Execution

  • Sends commands to robot controllers
  • Monitors execution
  • Provides feedback to user

Whisper: Robust Speech Recognition​

What is Whisper?​

Whisper is OpenAI's open-source speech recognition model trained on 680,000 hours of multilingual data.

Models:

  • tiny: 39M params, fastest
  • base: 74M params
  • small: 244M params
  • medium: 769M params
  • large: 1550M params, most accurate

Installation and Usage​

pip install openai-whisper

Basic Usage:

import whisper

# Load model
model = whisper.load_model("base")

# Transcribe audio
result = model.transcribe("audio.mp3")
print(result["text"])

Real-time Transcription:

import whisper
import pyaudio
import wave
import tempfile
import os

class RealtimeTranscriber:
    def __init__(self, model_size="base"):
        self.model = whisper.load_model(model_size)
        self.audio = pyaudio.PyAudio()

        self.chunk = 1024
        self.format = pyaudio.paInt16
        self.channels = 1
        self.rate = 16000
        self.record_seconds = 3

    def record_audio(self):
        stream = self.audio.open(
            format=self.format,
            channels=self.channels,
            rate=self.rate,
            input=True,
            frames_per_buffer=self.chunk
        )

        print("Listening...")
        frames = []

        for _ in range(0, int(self.rate / self.chunk * self.record_seconds)):
            data = stream.read(self.chunk)
            frames.append(data)

        stream.stop_stream()
        stream.close()

        return b''.join(frames)

    def transcribe(self, audio_data):
        # Save to temporary file
        with tempfile.NamedTemporaryFile(suffix=".wav", delete=False) as f:
            wf = wave.open(f.name, 'wb')
            wf.setnchannels(self.channels)
            wf.setsampwidth(self.audio.get_sample_size(self.format))
            wf.setframerate(self.rate)
            wf.writeframes(audio_data)
            wf.close()

            # Transcribe
            result = self.model.transcribe(f.name)
            os.unlink(f.name)

            return result["text"]

    def listen_and_transcribe(self):
        audio_data = self.record_audio()
        text = self.transcribe(audio_data)
        print(f"Transcribed: {text}")
        return text

# Usage
transcriber = RealtimeTranscriber(model_size="base")

while True:
    command = transcriber.listen_and_transcribe()
    if "stop listening" in command.lower():
        break

Using Gemini API​

import google.generativeai as genai
import os

genai.configure(api_key=os.getenv("GEMINI_API_KEY"))

class VoiceCommandProcessor:
    def __init__(self):
        self.model = genai.GenerativeModel('gemini-2.0-flash-exp')
        self.conversation_history = []

    def process_command(self, voice_input):
        # Add context about robot capabilities
        system_prompt = """You are a robot control assistant. The robot can:
        - Navigate to locations: "go to [location]"
        - Pick up objects: "pick up [object]"
        - Place objects: "place [object] on [location]"
        - Move joints: "move [joint] to [angle] degrees"
        - Execute sequences: "first [action], then [action]"

        Convert natural language commands into structured JSON actions.
        """

        user_message = f"{system_prompt}\n\nUser command: {voice_input}\n\nGenerate JSON action:"

        response = self.model.generate_content(user_message)

        return response.text

# Example usage
processor = VoiceCommandProcessor()

voice_input = "Please pick up the red cup and put it on the table"
action = processor.process_command(voice_input)
print(action)

Action Grounding​

import json
import numpy as np

class ActionGrounder:
    def __init__(self, robot_interface):
        self.robot = robot_interface
        self.object_positions = {}  # From perception system

    def ground_action(self, action_json):
        action = json.loads(action_json)

        for step in action.get("actions", []):
            action_type = step["type"]

            if action_type == "grasp":
                self.execute_grasp(step)
            elif action_type == "place":
                self.execute_place(step)
            elif action_type == "navigate":
                self.execute_navigate(step)
            elif action_type == "move_joint":
                self.execute_move_joint(step)

    def execute_grasp(self, action):
        object_name = action["object"]

        # Get object position from perception
        if object_name not in self.object_positions:
            print(f"Cannot find {object_name}")
            return False

        pos = self.object_positions[object_name]

        # Move to pre-grasp position
        pre_grasp = pos + np.array([0, 0, 0.1])
        self.robot.move_to(pre_grasp)

        # Approach object
        self.robot.move_to(pos)

        # Close gripper
        self.robot.gripper.close()

        # Lift
        self.robot.move_to(pos + np.array([0, 0, 0.1]))

        return True

    def execute_place(self, action):
        location = action["location"]
        target_pos = self.object_positions.get(location, np.array([0, 0, 0]))

        # Move to above placement location
        above_target = target_pos + np.array([0, 0, 0.2])
        self.robot.move_to(above_target)

        # Lower to placement height
        self.robot.move_to(target_pos + np.array([0, 0, 0.05]))

        # Open gripper
        self.robot.gripper.open()

        # Retract
        self.robot.move_to(above_target)

        return True

    def execute_navigate(self, action):
        target = action["target"]

        if target in self.object_positions:
            self.robot.navigate_to(self.object_positions[target])
        else:
            print(f"Unknown location: {target}")

    def execute_move_joint(self, action):
        joint_name = action["joint"]
        angle = action["angle"]

        self.robot.move_joint(joint_name, np.deg2rad(angle))

Complete Voice Control System​

import rclpy
from rclpy.node import Node
from std_msgs.msg import String
from geometry_msgs.msg import Twist
import whisper
import google.generativeai as genai
import json

class VoiceControlNode(Node):
    def __init__(self):
        super().__init__('voice_control_node')

        # Publishers
        self.cmd_pub = self.create_publisher(Twist, '/cmd_vel', 10)
        self.status_pub = self.create_publisher(String, '/robot/status', 10)

        # Initialize Whisper
        self.whisper_model = whisper.load_model("base")

        # Initialize Gemini
        genai.configure(api_key=os.getenv("GEMINI_API_KEY"))
        self.llm = genai.GenerativeModel('gemini-2.0-flash-exp')

        # Transcriber
        self.transcriber = RealtimeTranscriber()

        # Start voice control loop
        self.timer = self.create_timer(0.1, self.voice_control_loop)

        self.get_logger().info('Voice control node started')

    def voice_control_loop(self):
        # Listen for command
        command = self.transcriber.listen_and_transcribe()

        if not command:
            return

        self.get_logger().info(f'Heard: {command}')

        # Process with LLM
        action = self.process_command(command)

        # Execute action
        self.execute_action(action)

    def process_command(self, command):
        prompt = f"""Convert this robot command to JSON:
        "{command}"

        Available actions:
        - move_forward: {{"type": "move", "direction": "forward", "speed": 0.5}}
        - move_backward: {{"type": "move", "direction": "backward", "speed": 0.5}}
        - turn_left: {{"type": "turn", "direction": "left", "speed": 0.5}}
        - turn_right: {{"type": "turn", "direction": "right", "speed": 0.5}}
        - stop: {{"type": "stop"}}

        Return only JSON."""

        response = self.llm.generate_content(prompt)

        try:
            return json.loads(response.text)
        except:
            self.get_logger().error(f'Failed to parse: {response.text}')
            return {"type": "stop"}

    def execute_action(self, action):
        twist = Twist()

        if action["type"] == "move":
            if action["direction"] == "forward":
                twist.linear.x = action.get("speed", 0.5)
            elif action["direction"] == "backward":
                twist.linear.x = -action.get("speed", 0.5)

        elif action["type"] == "turn":
            if action["direction"] == "left":
                twist.angular.z = action.get("speed", 0.5)
            elif action["direction"] == "right":
                twist.angular.z = -action.get("speed", 0.5)

        elif action["type"] == "stop":
            twist.linear.x = 0.0
            twist.angular.z = 0.0

        self.cmd_pub.publish(twist)

        # Publish status
        status_msg = String()
        status_msg.data = f"Executing: {action['type']}"
        self.status_pub.publish(status_msg)

def main():
    rclpy.init()
    node = VoiceControlNode()
    rclpy.spin(node)
    node.destroy_node()
    rclpy.shutdown()

if __name__ == '__main__':
    main()

Handling Ambiguity and Safety​

Clarification Dialogs​

class SafetyLayer:
    def __init__(self):
        self.dangerous_actions = ["drop", "throw", "hit"]
        self.workspace_bounds = {
            "x": (-1.0, 1.0),
            "y": (-1.0, 1.0),
            "z": (0.0, 1.0)
        }

    def validate_action(self, action):
        # Check for dangerous keywords
        action_str = json.dumps(action).lower()
        for dangerous in self.dangerous_actions:
            if dangerous in action_str:
                return False, f"Action contains dangerous keyword: {dangerous}"

        # Check workspace bounds
        if "position" in action:
            pos = action["position"]
            for axis, (min_val, max_val) in self.workspace_bounds.items():
                if axis in pos:
                    if not (min_val <= pos[axis] <= max_val):
                        return False, f"{axis} position out of bounds"

        return True, "Action is safe"

    def request_clarification(self, command, llm):
        prompt = f"""The command "{command}" is ambiguous.
        Generate a clarifying question for the user."""

        response = llm.generate_content(prompt)
        return response.text

Multimodal Understanding​

import cv2
from PIL import Image

class MultimodalVLA:
    def __init__(self):
        self.llm = genai.GenerativeModel('gemini-2.0-flash-exp')
        self.transcriber = RealtimeTranscriber()

    def process_multimodal_command(self, image_path):
        # Get voice command
        voice_command = self.transcriber.listen_and_transcribe()

        # Load image
        image = Image.open(image_path)

        # Process with vision-language model
        prompt = f"""User command: "{voice_command}"

        Looking at the image, identify:
        1. Objects mentioned in the command
        2. Their locations
        3. Feasibility of the action
        4. Step-by-step plan

        Return JSON plan."""

        response = self.llm.generate_content([prompt, image])

        return json.loads(response.text)

# Usage
vla = MultimodalVLA()
plan = vla.process_multimodal_command("workspace.jpg")
